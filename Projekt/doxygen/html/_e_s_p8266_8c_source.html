32  {
33  TimeOut = 0;

35  return;
36  }
37

40  Response_Status = ESP8266_RESPONSE_WAITING;
41  }

45  _delay_ms(1);

53  if(!strncmp(RECEIVED_CRLF_BUF, _Expected_Response, EXPECTED_RESPONSE_LENGTH))
54  {
55  TimeOut = 0;
56  Response_Status = ESP8266_RESPONSE_FINISHED;
57  return;
58  }
